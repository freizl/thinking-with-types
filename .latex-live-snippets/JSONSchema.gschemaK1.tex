instance (KnownSymbol nm, KnownSymbol (ToJSONType a))
    => GSchema (M1 S ('MetaSel ('Just nm) _1 _2 _3)
                     (K1 _4 a)) where
  gschema = do
    emitRequired @nm  !\annotate{1}!
    pure . makePropertyObj @nm  !\annotate{2}!
         $ makeTypeObj @a
  {-# INLINE gschema #-}
