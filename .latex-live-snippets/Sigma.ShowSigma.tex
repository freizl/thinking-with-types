instance ( Dict1 Show (f :: k -> Type)
         , Show (Demote k)
         , SingKind k
         ) => Show (Sigma f) where
  show (Sigma sa fa) =
    case dict1 @Show @f sa of
      Dict -> mconcat
        [ "Sigma "
        , show $ fromSing sa
        , " ("
        , show fa
        , ")"
        ]
\end{code}
